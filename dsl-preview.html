DSL Configuration

How to Enable Features

Add the feature key to your CSS data config

Carousel

{
  "css": {
    "data": {
      "feature": "carousel",
      "container": {
        "css_class": "grid-3"
      }
    }
  }
}

Variants: carousel-cards, carousel-fade, carousel-cover

Tabs

{
  "css": {
    "data": {
      "feature": "tabs-pills",
      "container": {
        "css_class": "container"
      }
    }
  }
}

Variants: tabs-underline, tabs-vertical, tabs-steps

Accordion

{
  "css": {
    "data": {
      "feature": "accordion",
      "container": {
        "css_class": "container"
      }
    }
  }
}

Variants: accordion-bordered, accordion-flush
